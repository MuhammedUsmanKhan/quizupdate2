const LoadingArrItems = () => {

    for (let i = 0; i < lengthObj - 1; i++) {

            if (i == 0) {
                let QuesSpan = document.createElement(`span`)
                let txt = document.createTextNode(`${MCQ[x].Question}`)

                Label.appendChild(txt)
                Label.setAttribute(`id`, `Ans`)
